245template <typename T> inline Mat3<T> getRotationMatrix(T x_angle, T y_angle, T z_angle) {
246 const T sin_x = std::sin(x_angle);
247 const T cos_x = std::cos(x_angle);
248 const T sin_y = std::sin(y_angle);
249 const T cos_y = std::cos(y_angle);
250 const T sin_z = std::sin(z_angle);
251 const T cos_z = std::cos(z_angle);
252
253 const Mat3<T> rx = Mat3<T>({{1, 0, 0}, {0, cos_x, -sin_x}, {0, sin_x, cos_x}});
254 const Mat3<T> ry = Mat3<T>({{cos_y, 0, sin_y}, {0, 1, 0}, {-sin_y, 0, cos_y}});
255 const Mat3<T> rz = Mat3<T>({{cos_z, -sin_z, 0}, {sin_z, cos_z, 0}, {0, 0, 1}});
256
257 return rz * ry * rx;
258}
