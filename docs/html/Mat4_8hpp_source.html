317 static Mat4 LookAt(const Vec3<T>& eye, const Vec3<T>& center, const Vec3<T>& up) {
318 const Vec3<T> forward = (center - eye).normalized();
319 const Vec3<T> side = forward.cross(up).normalized();
320 const Vec3<T> upVector = side.cross(forward);
321
322 Mat4 lookAtMatrix;
323 lookAtMatrix.m[0][0] = side.x;
324 lookAtMatrix.m[0][1] = side.y;
325 lookAtMatrix.m[0][2] = side.z;
326 lookAtMatrix.m[0][3] = -(side.x * eye.x + side.y * eye.y + side.z * eye.z);
327
328 lookAtMatrix.m[1][0] = upVector.x;
329 lookAtMatrix.m[1][1] = upVector.y;
330 lookAtMatrix.m[1][2] = upVector.z;
331 lookAtMatrix.m[1][3] = -(upVector.x * eye.x + upVector.y * eye.y + upVector.z * eye.z);
332
333 lookAtMatrix.m[2][0] = -forward.x;
334 lookAtMatrix.m[2][1] = -forward.y;
335 lookAtMatrix.m[2][2] = -forward.z;
336 lookAtMatrix.m[2][3] = (forward.x * eye.x + forward.y * eye.y + forward.z * eye.z);
337
338 lookAtMatrix.m[3][0] = 0.0;
339 lookAtMatrix.m[3][1] = 0.0;
340 lookAtMatrix.m[3][2] = 0.0;
341 lookAtMatrix.m[3][3] = 1.0;
342
343 return lookAtMatrix;
344 }

372 static Mat4 Orthographic(T left, T right, T bottom, T top, T near, T far) {
373 Mat4 ortho;
374 ortho.m[0][0] = 2 / (right - left);
375 ortho.m[1][1] = 2 / (top - bottom);
376 ortho.m[2][2] = -2 / (far - near);

394 static Mat4 Perspective(T fovY, T aspect, T near, T far) {
395 T invTanHalfFovY = 1.0 / std::tan(fovY / 2);
396 Mat4 persp;
397 persp.m[0][0] = invTanHalfFovY / aspect;
398 persp.m[1][1] = invTanHalfFovY;
399 persp.m[2][2] = -(far + near) / (far - near);
400 persp.m[2][3] = -(2 * far * near) / (far - near);
401 persp.m[3][2] = -1.0;
402 persp.m[3][3] = 0.0;
403
404 return persp;
