322 static Mat4 LookAt(const Vec3<T>& eye, const Vec3<T>& center, const Vec3<T>& up) {
323 const Vec3<T> forward = (center - eye).normalized();
324 const Vec3<T> side = forward.cross(up).normalized();
325 const Vec3<T> up_vector = side.cross(forward);
326
327 Mat4 look_at_matrix;
328 look_at_matrix.m[0][0] = side.x;
329 look_at_matrix.m[0][1] = side.y;
330 look_at_matrix.m[0][2] = side.z;
331 look_at_matrix.m[0][3] = -(side.x * eye.x + side.y * eye.y + side.z * eye.z);
332
333 look_at_matrix.m[1][0] = up_vector.x;
334 look_at_matrix.m[1][1] = up_vector.y;
335 look_at_matrix.m[1][2] = up_vector.z;
336 look_at_matrix.m[1][3] = -(up_vector.x * eye.x + up_vector.y * eye.y + up_vector.z * eye.z);
337
338 look_at_matrix.m[2][0] = -forward.x;
339 look_at_matrix.m[2][1] = -forward.y;
340 look_at_matrix.m[2][2] = -forward.z;
341 look_at_matrix.m[2][3] = (forward.x * eye.x + forward.y * eye.y + forward.z * eye.z);
342
343 look_at_matrix.m[3][0] = 0.0;
344 look_at_matrix.m[3][1] = 0.0;
345 look_at_matrix.m[3][2] = 0.0;
346 look_at_matrix.m[3][3] = 1.0;
347
348 return look_at_matrix;
349 }

378 static Mat4 Orthographic(T left, T right, T bottom, T top, T near, T far) {
379 Mat4 ortho;
380 ortho.m[0][0] = 2 / (right - left);
381 ortho.m[1][1] = 2 / (top - bottom);

400 static Mat4 Perspective(T fovY, T aspect, T near, T far) {
401 T inv_tan_half_fov_y = 1.0 / std::tan(fovY / 2);
402 Mat4 persp;
403 persp.m[0][0] = inv_tan_half_fov_y / aspect;
404 persp.m[1][1] = inv_tan_half_fov_y;

406 persp.m[2][3] = -(2 * far * near) / (far - near);
